⚙️ Flask Integration — 4 Lines

from secureshield import scan_url
from flask import Flask, request

app = Flask(__name__)

@app.before_request
def protect():
    attack = scan_url(request.url)
    if attack:
        return f"⚠️ Blocked: {attack}", 403
